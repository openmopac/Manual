zoom 0

#   The phenolic hydrogen on Tyr29 is moved to the ortho carbon.

#   Create a bond between the hydrogen and the ortho carbon,
#   and delete the bond from the hydrogen to the oxygen
#
connect (atomno=418) (atomno=406) single
connect (atomno=418) (atomno=409) delete
delay 2
minimize SELECT (29)
